This monorepo contains the frontend applications, backend microservices, and infrastructure configurations for the Financial Investment Platform (Crypto, Stocks, Gold, and Asset Management).

      - platform-network

  # Redis for caching and fast lookups (Session data, Market Data cache)

    image: redis:7-alpine
    container_name: platform-redis
    ports:
      - "6379:6379"
    volumes:
      - redis_data:/data
    networks:
      - platform-network

  # Zookeeper (Required for Kafka)
  zookeeper:
    image: confluentinc/cp-zookeeper:7.4.0
    container_name: platform-zookeeper
    environment:
      ZOOKEEPER_CLIENT_PORT: 2181
      ZOOKEEPER_TICK_TIME: 2000
    ports:
      - "2181:2181"
    networks:
      - platform-network

  # Kafka for event-driven architecture (Order matching, Market data streams)
  kafka:
    image: confluentinc/cp-kafka:7.4.0
    container_name: platform-kafka
    depends_on:
      - zookeeper

      - "9092:9092"
    environment:
      KAFKA_BROKER_ID: 1

      KAFKA_INTER_BROKER_LISTENER_NAME: PLAINTEXT
      KAFKA_OFFSETS_TOPIC_REPLICATION_FACTOR: 1
